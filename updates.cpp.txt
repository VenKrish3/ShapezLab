#include <iostream>
#include <cassert>
#include <cmath>
#include <iomanip>

using namespace std;

class Shape
{
	public:
		virtual	float getArea() = 0;
		virtual	float getPerimeter() = 0;
};

// Wrong Triangle
class RightTriangle : public Shape {
	public:
		RightTriangle(float, float);
		float getArea();
		float getPerimeter();
	private:
		float height;
		float base;
};
RightTriangle::RightTriangle(float base, float height) {
	this->base = base;
	this->height = height;
}
float RightTriangle::getArea() {
	return (base * height) / 2;
};
float RightTriangle::getPerimeter() {
	return (base + height) + sqrt((base * base) + (height * height));
};

//iosooleces Triangle
class IsoscelesTriangle : public RightTriangle {
    public:
		IsoscelesTriangle(float);
		float getArea();
		float getPerimeter();
	private:
		float side;
};
IsoscelesTriangle::IsoscelesTriangle(float side) {
	this->side = side;
}
float IsoscelesTriangle::getArea() {
	return (side * side) / 2;
};
float IsoscelesTriangle::getPerimeter() {
	return (side + side) + sqrt((side * side) + (side * side));
};


// Rectangale
class Rectangle : public Shape {
	public:
		Rectangle(float, float);
		float getArea();
		float getPerimeter();
	private:
		float length;
		float width;
};
Rectangle::Rectangle(float length, float width) {
	this->length = length;
	this->width = width;
}
float Rectangle::getArea() {
	return width * length;
};
float Rectangle::getPerimeter() {
	return (width * 2) + (length * 2);
};

// Circle
class Circle : public Shape {
	public:
		Circle(float);
		float getArea();
		float getPerimeter();
	private:
		int radius;	
		float pi = 3.14;
};
Circle::Circle(float radius) {
	this->radius = radius;
}
float Circle::getArea() {
	return pow((pi*radius),2);
};
float Circle::getPerimeter() {
	return 2*pi*radius;
};

// main
int main() {
    cout << fixed << setprecision(2);
    int x = 3, y = 3;
	Circle circle(x);
	Rectangle rect(x, y);
	RightTriangle trang(x, y);
	
	circle.getArea();
	rect.getArea();
	trang.getPerimeter();
	assert(circle.getArea() <= 88.74 && circle.getArea() >= 88.72);
	assert(rect.getArea() == 9);
	assert(trang.getArea() == 4.5);
	assert(circle.getPerimeter() <= 18.85 && circle.getPerimeter() >= 18.83);
	assert(rect.getPerimeter() == 12);
	assert(trang.getPerimeter() <= 10.25 && trang.getPerimeter() >= 10.23);
}
